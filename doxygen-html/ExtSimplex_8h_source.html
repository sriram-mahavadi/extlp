00119     unsigned int update_simplex_iteration() {
00120         // Simple reference to make coding easy
00121         ExtMatrixA& A = extDataSet.A;
00122         ExtMatrixBInverse &b_inverse = extDataSet.BInverse;
00123         bool is_min = extDataSet.get_objective_sense_bool();
00124         //! Finding the Entering Variable - k 
00125         //! Note: k index is with respect to Matrix A
00126 
00127         //! Calculating w = < Cb x Binverse >
00128         PackedVector w(get_basis_dimension());
00129         ExtVectorUtil::store_dot_product(m_vct_obj, b_inverse, w);
00130 
00131         // Getting j for which cj-zj gives max positive value => k
00132         unsigned int k = -1;
00133         REAL max_reduced_cost = -INFINITY_VALUE;
00134         for (unsigned int j = 0; j < A.get_columns_count(); j++) {
00135             ExtMatrixA::ColAttr col_attr = A.get_col_attr(j);
00136             if (!col_attr.get_is_basic_col()) {
00137                 // If the column is a non basic column
00138                 PackedVector a_col_vector(get_basis_dimension());
00139                 A.store_column(j, a_col_vector, false);
00140                 REAL cj = col_attr.get_objective_value(); // cj value
00141                 if (is_min)cj *= -1;
00142                 REAL zj = 0.0F;
00143                 ExtVectorUtil::store_dot_product(w, a_col_vector, zj);
00144                 REAL reduced_cost = cj - zj;
00145                 // Always pick the first max postive cj-zj - Bland Method
00146                 if (reduced_cost > max_reduced_cost) {
00147                     max_reduced_cost = reduced_cost;
00148                     k = j;
00149                 }
00150                 //                DEBUG("Col: " << j << ", Reduced Cost: " << reduced_cost);
00151             }
00152         }
00153         //        debug_objective_values();
00154         //        debug_b_inverse();
00155         //        debug_w(w);
00156         if (max_reduced_cost == 0.0F) return SIMPLEX_CJZJ_ZERO;

00158 
00159         //! Getting column-k of Matrix A
00160         PackedVector ak_vector(get_basis_dimension());
00161         A.store_column(k, ak_vector, false);
00162 
00163         //! Finding the Leaving Variable - r based on - min br/ykr.
00164         //! Note: r index is with respect to MatrixBInverse
00165         //! Finding yk
00166         SimpleVector<REAL> y(get_basis_dimension());
00167         ExtVectorUtil::store_dot_product(b_inverse, ak_vector, y);
00168         //        debug_y(y);
00169         unsigned int r; // leaving variable
00170         REAL min_ratio = -1.0F;
00171         //! Finding r - leaving variable 
00172         for (unsigned int row_index = 0; row_index < y.get_size(); row_index++) {
00173             if (y[row_index] == 0.0F)continue; // Ignoring when yk=0
00174             if (m_vct_rhs[row_index] >= 0.0F && y[row_index] < 0.0F)continue; // Ignoring when ration<0
00175             REAL ratio = m_vct_rhs[row_index] / y[row_index];
00176             if ((min_ratio == -1.0F && ratio >= 0.0F) ||
00177                     (ratio >= 0.0F && ratio < min_ratio)) {
00178                 min_ratio = ratio;
00179                 r = row_index;
00180             }
00181             // Elimintating Cycles in Simplex - Bland Method
00182             if (ratio >= 0.0F && ratio == min_ratio) {
00183                 // Already existing leaving variable may have a greater index value, then replace it
00184                 unsigned int a_r_index = b_inverse.get_col_attr(r).get_a_col_index();
00185                 unsigned int a_row_index = b_inverse.get_col_attr(row_index).get_a_col_index();
00186                 if (a_r_index > a_row_index) {
00187                     r = row_index;
00188                 }
00189             }
00190         }
00191         if (min_ratio == -1.0F) {
00192             return SIMPLEX_INVALID_MIN_RATIO;
00193         }

00197         //        debug_y(y);
00198         //        debug_rhs();
00199         // Getting the Eta Vector for updating data structures for next iteration
00200         EtaVector eta_vector(r, y);
00201         //        debug_eta(eta_vector);
00202 
00203         //        DEBUG_SIMPLE("*************************** BEFORE ITERERATION " << m_iteration_count << " ******************************************");
00204         //        DEBUG_SIMPLE("             **************************************************");
00205         //        debug_objective_values();
00206         //        debug_w(w);
00207         //        debug_ak(ak_vector);
00208         //        debug_rhs();
00209         //        debug_y(y);
00210         //        debug_eta(eta_vector);
00211         //        debug_b_inverse();
00212 
00213         // Updating A Matrix for setting cols to basic 
00214         unsigned int a_r_index = b_inverse.get_col_attr(r).get_a_col_index();
00215         ExtMatrixA::ColAttr a_r_col_attr = A.get_col_attr(a_r_index);
00216         a_r_col_attr.set_is_basic_col(false);
00217         A.set_col_attr(a_r_index, a_r_col_attr);
00218         ExtMatrixA::ColAttr a_k_col_attr = A.get_col_attr(k);
00219         a_k_col_attr.set_is_basic_col(true);
00220         A.set_col_attr(k, a_k_col_attr);
00221 
00222         // Updating B inverse
00223         ExtVectorUtil::update_basis_inverse(eta_vector, b_inverse);
00224         ExtMatrixBInverse::ColAttr col_attr = b_inverse.get_col_attr(r);
00225         col_attr.set_a_col_index(k);
00226         b_inverse.set_col_attr(r, col_attr);
00227 
00228         //        debug_base_col_indices();
00229         // Updating Base Col indices
00230         m_base_col_indices[r] = k;
00231 
00232         // Updating the objective values
00233         REAL k_coef_value = A.get_col_attr(k).get_objective_value();
00234         if (is_min)k_coef_value *= -1;
00235         m_vct_obj[r] = k_coef_value;

00244         //        DEBUG_SIMPLE("*************************** AFTER ITERERATION " << m_iteration_count-1 << " ******************************************");
00245         //        DEBUG_SIMPLE("             **************************************************");
00246         //        debug_objective_values();
00247         //        debug_rhs();
00248         //        debug_b_inverse();
00249         //        debug_base_col_indices();
00250         return SIMPLEX_CJZJ_POSITIVE;
